let scaneach k f z elems = 
    let c = ref 0
    seq {
        let zref = ref z
        c := (!c + 1)%k
        if !c = 0 then
            yield !zref
        for e in elems do
            zref := f !zref e
            c := (!c + 1)%k
            if !c = 0 then
                yield !zref
    }
