QR

A simple canonical use

Many operations on list can be seen as a fold, where a state pushed in, then mixed sequentially through the elements, to give a result in the end. A "right fold" or "fold back" gives will access the deepest element first, by recursing from within the reduce function. As you can see here, it first recurses, then applies the reduction.

let GenerateRandomDenseMatrix  (row,col) = 
     let normal = Normal(RandomSource = new MersenneTwister(1))
     let matrixA = new DenseMatrix(row, col)
     for i = 0  to row - 1 do
        for j = 0 to col - 1  do
            matrixA.[i, j] <- normal.Sample()
     matrixA;


type QR = {Q : Matrix<double>; R :Matrix<double> } 
 
let myQRGS (mat:Matrix<double>) = 
    let n = mat.RowCount
    let res = { Q= DenseMatrix.zeroCreate n n  ; R = DenseMatrix.zeroCreate n n }
    let newCurrent = DenseVector.zeroCreate n
    
    let rec computekCol k = 
        if k = n then ()
        else
         let current = mat.Column(k)

         newCurrent.SetValues(current.ToArray())
            
         //we remove from this value what was already known
         //for all the previously 
         for i = 0 to k - 1 do                                     //(A_k . Q_i)
            let formerQi = res.Q.Column(i)
            let proj_coef_i = current.DotProduct(formerQi)             //how much do we already know 
            do res.R.At(i,k, proj_coef_i)                              //store the coef in the k column, current row
            do newCurrent.Subtract(proj_coef_i * formerQi, newCurrent) //we remove what we already have
            
         // we normalize what is left, as the linear component appears in the dotproduct
         // and is already stored in Q
         let normnewStuff = newCurrent.Norm(2.)
         do newCurrent.Divide(normnewStuff, newCurrent)

         //we store the normalized new column, which is jsut the new stuff
         res.Q.SetColumn(k, newCurrent)
         //we store the norm
         res.R.At(k,k, normnewStuff)  
         computekCol (k+1)
    do computekCol 0
    res
    
          

let QRworks(row,col) =
    let matrixA =  GenerateRandomDenseMatrix  (row,col)
    let factorQR = matrixA.QR();
    ()
